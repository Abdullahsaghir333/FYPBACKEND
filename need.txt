**AGENT PROMPT 3 — EMOTION-AWARE TEACHING**

```python
# Add to _monitor() return dict:
def _detect_expression(self, lm, ear, delta_pitch) -> str:
    # Brow furrow proxy: distance between landmarks 9 (forehead) and 8 (nose bridge)
    brow_dist = abs(lm[9].y - lm[8].y)
    # Confusion: head slightly tilted + brow compressed
    if delta_pitch > 10 and brow_dist < 0.08:
        return "CONFUSED"
    # Boredom: low EAR but not sleeping, flat head, extended duration
    if ear < (self.baseline_ear - 0.02) and delta_pitch < 5:
        return "BORED"
    return "NEUTRAL"
```
